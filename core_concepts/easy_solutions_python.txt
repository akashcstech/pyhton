37.Create a class Animal with a method sound(). Create a subclass Dog that overrides sound() and returns "Bark".
------------------------------------------------------------------------------------------------------------------
class Animal:
    def __init__(self, pet):
        self.pet = pet
    
    def sound(self):
        print("Meow")


class subclass(Animal):
    def sound(self):
        print("Bark")


obj1 = Animal("Dog")
obj1.sound()

obj2 = subclass("Dog")
obj2.sound()

38.Create two classes Cat and Cow, each with a method speak(). Call speak() for both using the same function (polymorphism).
-----------------------------------------------------------------------------------------------------------------------------
class cat:
    def __init__(self, cat):
        self.cat = cat

    def speak():
        print("meow")

class cow:
    def __init__(self, cow):

        self.cow = cow 

    def speak():
        print("Moo")
        

obj1 = cat
obj2 = cow
print(obj1.speak())
print(obj2.speak())
